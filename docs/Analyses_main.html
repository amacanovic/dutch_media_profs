fields <- fields[!is.na(fields)]
field_comparisons <- data.frame(matrix(NA, ncol = 7, nrow = length(fields)))

for (i in 1:length(fields)){
  field <- fields[i]
  data <- filter(sources_check,

  field_comparisons[i, 1] <- field
